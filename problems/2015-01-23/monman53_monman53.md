```haskell
lsystem :: (Char -> [Char]) -> [Char] -> Int -> [Char]
lsystem rules start n
  | n == 0    = start
  | otherwise = replace' rules (lsystem' rules start (n-1))
  where replace' _ []         = []
        replace' r (c:after)  = (r c) ++ (replace' r after)
```
